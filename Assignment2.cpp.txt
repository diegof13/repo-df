#include<iostream>
#include<cstdlib>
#include<stdlib.h>
#include<string>
#include<stdbool.h>
using namespace std;

class Rational_number {
	private:
	int a;
	int b;
	int c;
	int d;
public:
	Rational_number(int a, int b, int c, int d) {
		this->a = a;
		this->b = b;
		this->c = c;
		this->d = d;
	}
	void add(int sum = 0) {
		cout << "\nThe sum of the 2 numbers is: ";//((a * d) + (b * c)) / d * c;
		sum = (a * d) + (b * c);
		cout << a << "/" << b << " + " << c << "/" << d << " = " << sum << "/" << b * d << endl;
	}
	void sub(int res = 0) {
		cout << "\nThe substraction of the 2 numbers is: ";//((a * d) - (b * c)) / d * c;
		res = (a * d) - (b * c);
		cout << a << "/" << b << " - " << c << "/" << d << " = " << res << "/" << b * d << endl;
	}
	void mult(int prod = 1) {
		cout << "\nThe product of the 2 numbers is: ";
		prod = a * c;
		cout << a << "/" << b << " * " << c << "/" << d << " = " << prod << "/" << b * d << endl;
	}
	void div(int ov = 1) {
		cout << "\nThe division of the 2 numbers is: ";
		ov = b * c;
		cout << "(" << a << "/" << b << ")" << "/" << "(" << c << "/" << d << ")" << " = " << a * d << "/" << ov << endl;
	}
	int get_1st_number() {
		cout << "The first number is:";
		cout << a << "/" << b << endl;
		cout << "The second number is:";
		cout << c << "/" << d << endl;
		return a, b, c, d;
	}
	bool less_than(int a, int b, int c, int d) {
		float x1, x2;
		x1 = a / b;
		x2 = c / d;
		if (x1 < x2) {
			cout << "\nBool value :" << true << endl;
			cout << endl << a << "/" << b << " is less than " << c << "/" << d << endl;
		}
		else {
			cout << "\nBool value :" << false << endl;
			cout << endl << a << "/" << b << " is not less than " << c << "/" << d << endl;
		}
		return x1, x2;
	}
};

int main() {
	int n1 = 3, d1 = 4, n2 = 3, d2 = 2;
	Rational_number number1(n1, d1, n2, d2);//Constructor 
	number1.get_1st_number();//  3/4  and  3/2
	number1.add();//   18/8
	number1.sub();//   -6/8
	number1.mult();//  9/8
	number1.div();//   6/12
	number1.less_than(n1,d1,n2,d2);//1

	return 0;
}
