#include<iostream>
#include<cstdlib>
#include<stdlib.h>
#include<string>
using namespace std;

class Rational_number {
	private:
	int a;
	int b;
public:
	Rational_number(int a, int b) {
		this->a = a;
		this->b = b;
	}
	void add(int sum = 0) {
		cout << "\nThe sum of the 2 numbers is: ";
		sum = a + b;
		cout << sum;
	}
	void sub(int res = 0) {
		cout << "\nThe substraction of the 2 numbers is: ";
		res = a - b;
		cout << res;
	}
	void mult(int prod = 1) {
		cout << "\nThe product of the 2 numbers is: ";
		prod = a * b;
		cout << a << "*" << b << " = " << prod;
	}
	void div(int ov = 1) {
		cout << "\nThe division of the 2 numbers is: ";
		cout << a << "/" << b;
	}
	int get_1st_number() {
		cout << "The first number is:";
		cout << a << "/" << b << endl;
		return a, b;
	}
};

int main() {
	Rational_number number1(5, 4);//Constructor 
	number1.get_1st_number();//  5/4
	number1.add();//   9
	number1.sub();//   1
	number1.mult();//  20
	number1.div();//   5/4

	return 0;
}
