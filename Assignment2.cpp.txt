#include<iostream>
#include<cstdlib>
#include<stdlib.h>
#include<string>
using namespace std;

class Rational_number {
	private:
	int a;
	int b;
	int c;
	int d;
public:
	Rational_number(int a, int b, int c, int d) {
		this->a = a;
		this->b = b;
		this->c = c;
		this->d = d;
	}
	void add(int sum = 0) {
		cout << "\nThe sum of the 2 numbers is: ";//((a * d) + (b * c)) / d * c;
		sum = (a * d) + (b * c);
		cout << a << "/" << b << " + " << c << "/" << d << " = " << sum << "/" << b * d << endl;
	}
	void sub(int res = 0) {
		cout << "\nThe substraction of the 2 numbers is: ";//((a * d) - (b * c)) / d * c;
		res = (a * d) - (b * c);
		cout << a << "/" << b << " - " << c << "/" << d << " = " << res << "/" << b * d << endl;
	}
	void mult(int prod = 1) {
		cout << "\nThe product of the 2 numbers is: ";
		prod = a * c;
		cout << a << "/" << b << " * " << c << "/" << d << " = " << prod << "/" << b * d << endl;
	}
	void div(int ov = 1) {
		cout << "\nThe division of the 2 numbers is: ";
		ov = b * c;
		cout << "(" << a << "/" << b << ")" << "/" << "(" << c << "/" << d << ")" << " = " << a * d << "/" << ov << endl;
	}
	int get_1st_number() {
		cout << "The first number is:";
		cout << a << "/" << b << endl;
		cout << "The second number is:";
		cout << c << "/" << d << endl;
		return a, b, c, d;
	}
};

int main() {
	Rational_number number1(5, 4, 1, 2);//Constructor 
	number1.get_1st_number();//  5/4  and  1/2
	number1.add();//   14/8
	number1.sub();//   6/8
	number1.mult();//  5/8
	number1.div();//   10/4

	return 0;
}
