#include<iostream>

using namespace std;

class Polynomial {
public:
	int n, * Poly1, * Poly2, * Poly3;
	Polynomial() {
		Poly1 = new int[n];
		Poly2 = new int[n];
		Poly3 = new int[n];
	}
	void ask_data() {//Asking data
		cout << "Enter size of polynomials: "; cin >> n;
		cout << "  (1ST POLYNOMIAL)";
		for (int i = 0;i < n;i++) {
			cout << "  \nEnter coefficient [" << i << "]: "; cin >> Poly1[i];
		}
		cout << "  (2ND POLYNOMIAL)";
		for (int i = 0;i < n;i++) {
			cout << "  \nEnter coefficient [" << i << "]: "; cin >> Poly2[i];
		}
	}
	void show_polynomials() {//showing polynomials
		cout << "P1 = ";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly1[i] << " ";
		}cout << endl;
		cout << "P2 = ";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly2[i] << " ";
		}
	}
	void addition() {
		for (int i = 0;i < n;i++) {
			Poly3[i] = Poly1[i] + Poly2[i];
		}cout << endl;
		cout << "p1+p2=";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly3[i] << " ";
		}
	}
	void subtraction() {
		for (int i = 0;i < n;i++) {
			Poly3[i] = Poly1[i] - Poly2[i];
		}cout << endl;
		cout << "p1-p2=";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly3[i] << " ";
		}
	}
	~Polynomial() {
		delete[] Poly1;
		delete[] Poly2;
		delete[] Poly3;
	}
};

int main() {
	Polynomial p;
	p.ask_data();
	p.show_polynomials();
	p.addition();
	p.subtraction();
	return 0;
}
