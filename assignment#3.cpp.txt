#include<iostream>

using namespace std;

class Poly {
private:
	string Polynomial1, Polynomial2;
	int* arr1, * arr2, * arr3, n;
public:
	Poly(string Polynomial1, string Polynomial2) {
		this->Polynomial1 = Polynomial1;
		this->Polynomial2 = Polynomial2;
		arr1 = new int;
		arr2 = new int;
		arr3 = new int;
	}
	Poly() {
		arr1 = new int;
		arr2 = new int;
		arr3 = new int;
	}
	string get_name1() {
		return Polynomial1;
	}
	string get_name2() {
		return Polynomial2;
	}
	void set_name1(string Polynomial1) {
		this->Polynomial1 = Polynomial1;
	}
	void set_name2(string Polynomial2) {
		this->Polynomial2 = Polynomial2;
	}
	void askdata1() {
		cout << "\nEnter size: "; cin >> n;
		for (int i = 0;i < n;i++) {
			cout << "\nType [" << i << "]: "; cin >> arr1[i];
		}
		//arr1[0] = 7;
		//arr1[1] = 8;
		//arr1[2] = 9;
	}
	void askdata2() {
		cout << "\nEnter size: "; cin >> n;
		for (int i = 0;i < n;i++) {
			cout << "\nType [" << i << "]: "; cin >> arr2[i];
		}
	
		//arr2[0] = 3;
		//arr2[1] = 2;
		//arr2[2] = 3;
	}
	void get_array1() {
		for (int i = 0;i < 3;i++) {
			cout << arr1[i] << " ";
		}
	}
	void get_array2() {
		for (int i = 0;i < 3;i++) {
			cout << arr2[i] << " ";
		}
	}
	void add() {
		for (int i = 0;i < 3;i++) {
			arr3[i] = arr1[i] + arr2[i];
		}cout << "SUM= ";
		for (int i = 0;i < 3;i++) {
			cout << " " << arr3[i] << " ";
		}
	}
	void subs() {
		for (int i = 0;i < 3;i++) {
			arr3[i] = arr1[i] - arr2[i];
		}cout << "SUBS= ";
		for (int i = 0;i < 3;i++) {
			cout << " " << arr3[i] << " ";
		}
	}
	void prod() {
		cout << "PROD= ";
		for (int i = 0;i < 3;i++) {
			for (int j = 0;j < 3;j++) {
				arr3[i + j] = arr1[i] * arr2[i];
			}
		}
		for (int i = 0;i < 3;i++) {
			for (int j = 0;j < 3;j++) {
				cout << arr3[i + j] << " ";
			}
		}
	}
	~Poly() {
		delete[] arr1;
		//arr1 = NULL;
		delete[] arr2;
		//arr2 = NULL;
		delete[] arr3;
		//arr3 = NULL;
	}

};

int main() {
	Poly p("1st Polynimial", "2nd Polynomial");
	cout << p.get_name1() << endl;
	cout << p.get_name2() << endl;
	p.askdata1();
	p.askdata2();
	cout << endl;
	p.get_array1();
	cout << endl;
	p.get_array2();
	cout << endl;
	p.add();
	cout << endl;
	p.subs();
	cout << endl;
	p.prod();

	return 0;
}
