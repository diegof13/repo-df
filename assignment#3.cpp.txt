class Poly {
private:
	string Polynomial1, Polynomial2;
	int arr1[6], arr2[6], arr3[100], x, i, j;
public:
	Poly(string Polynomial1, string Polynomial2) {

		this->Polynomial2 = Polynomial2;

	string get_name1() {
		return Polynomial1;
	}
	string get_name2() {
		return Polynomial2;
	}
	void set_name1(string Polynomial1) {
		this->Polynomial1 = Polynomial1;

	void set_name2(string Polynomial2) {

	}
	void askdata1() {
		arr1[0] = 5;

		arr1[2] = 6;
	}
	void askdata2() {
		arr2[0] = 1;
		arr2[1] = 2;
		arr2[2] = 3;
	}
	void get_array1() {
		for (i = 0;i < 3;i++) {
			cout << arr1[i] << " ";

	}

		for (i = 0;i < 3;i++) {
			cout << arr2[i] << " ";
		}
	}
	void add() {
		for (i = 0;i < 3;i++) {
			arr3[i] = arr1[i] + arr2[i];
		}cout << "SUM= ";
		x = i - 1;
		for (j = x;j >= 0;j--) {
			if (j == 0) {
				cout << arr3[j];
				break;
			}
			cout << arr3[j] << "(x^" << x << ")" << "+";
			x--;
		}
	}
	void subs() {
		for (i = 0;i < 3;i++) {
			arr3[i] = arr1[i] - arr2[i];
		}cout << "SUBS= ";
		x = i - 1;
		for (j = x;j >= 0;j--) {
			if (j == 0) {
				cout << arr3[j];
				break;
			}
			cout << arr3[j] << "(x^" << x << ")" << "+";
			x--;
		}
	}
	void prod() {
		cout << "PROD= ";
		for (i = 0;i < 3;i++) {
			for (j = 0;j < 3;j++) {
				arr3[i+j] = arr1[i] * arr2[i];
			}
		}
		//for (i = 0;i < 3;i++) {

				//cout << arr3[i + j] << " ";
			//}
		x = i - 1;
		for (j = x;j >= 0;j--) {
			if (j == 0) {
				cout << arr3[j];
				break;
			}
			cout << arr3[j] << "(x^" << x << ")" << "+";
			x--;
		}
	}
};
