#include<iostream>

using namespace std;

class Poly {
private:
	string Polynomial1, Polynomial2;
	int* arr1, * arr2, * arr3, x, i, j;
public:
	Poly(string Polynomial1, string Polynomial2) {
		this->Polynomial1 = Polynomial1;
		this->Polynomial2 = Polynomial2; 
		arr1 = new int[3];
		arr1 = new int[3];
		arr1 = new int[10];
	}
	Poly() {
		arr1 = new int[3];
		arr1 = new int[3];
		arr1 = new int[10];
	}
	string get_name1() {
		return Polynomial1;
	}
	string get_name2() {
		return Polynomial2;
	}
	void set_name1(string Polynomial1) {
		this->Polynomial1 = Polynomial1;
	}
	void set_name2(string Polynomial2) {
		this->Polynomial2 = Polynomial2;
	}
	void getdata1() {
		for (i = 0;i < 3;i++) {
			cout << "Enter [" << i << "]: "; cin >> arr1[i];
		}
		//arr1[0] = 5;
		//arr1[1] = 4;
		//arr1[2] = 6;
	}
	void getdata2() {
		for (i = 0;i < 3;i++) {
			cout << "Enter [" << i << "]: "; cin >> arr2[i];
		}
		//arr2[0] = 1;
		//arr2[1] = 2;
		//arr2[2] = 3;
	}
	void get_array1() {
		for (i = 0;i < 3;i++) {
			cout << arr1[i] << " ";
		}
	}
	void get_array2() {
		for (i = 0;i < 3;i++) {
			cout << arr2[i] << " ";
		}
	}
	void add() {
		for (i = 0;i < 3;i++) {
			arr3[i] = arr1[i] + arr2[i];
		}cout << "SUM= ";
		x = i - 1;
		for (j = x;j >= 0;j--) {
			if (j == 0) {
				cout << arr3[j];
				break;
			}
			cout << arr3[j] << "(x^" << x << ")" << "+";
			x--;
		}
	}
	void subs() {
		for (i = 0;i < 3;i++) {
			arr3[i] = arr1[i] - arr2[i];
		}cout << "SUBS= ";
		x = i - 1;
		for (j = x;j >= 0;j--) {
			if (j == 0) {
				cout << arr3[j];
				break;
			}
			cout << arr3[j] << "(x^" << x << ")" << "+";
			x--;
		}
	}
	void prod() {
		cout << "PROD= ";
		for (i = 0;i < 3;i++) {
			for (j = 0;j < 3;j++) {
				arr3[i+j] = arr1[i] * arr2[i];
			}
		}
		//for (i = 0;i < 3;i++) {
			//for (j = 0;j < 3;j++) {
				//cout << arr3[i + j] << " ";
			//}
		x = i - 1;
		for (j = x;j >= 0;j--) {
			if (j == 0) {
				cout << arr3[j];
				break;
			}
			cout << arr3[j] << "(x^" << x << ")" << "+";
			x--;
		}
	}
	Poly(Poly& f) {
		this->arr1 = new int[3];
		for (i = 0;i < 3;i++) {
			this->arr1[i] = f.arr1[i];
		}
		this->arr2 = new int[3];
		for (i = 0;i < 3;i++) {
			this->arr2[i] = f.arr2[i];
		}
		this->arr3 = new int[10];
		for (i = 0;i < 10;i++) {
			this->arr3[i] = f.arr3[i];
		}
	}
	void operator=(Poly x) {
		for (i = 0;i < 3;i++) {
			this->arr1[i] = x.arr1[i];
		}
	}
	void operator=(Poly y) {
		for (i = 0;i < 3;i++) {
			this->arr2[i] = y.arr2[i];
		}
	}
	void operator=(Poly z) {
		for (i = 0;i < 10;i++) {
			this->arr3[i] = z.arr3[i];
		}
	}
	~Poly() {
		delete[]arr1;
		delete[]arr2;
		delete[]arr3;
	}
};

int main() {
	Poly p("COEFFICIENTS 1st Polynimial", "COEFFICIENTS 2nd Polynomial");
	cout << p.get_name1() << endl;
	cout << p.get_name2() << endl;
	p.getdata1();
	p.getdata2();
	p.get_array1();
	cout << endl;
	p.get_array2();
	cout << endl;
	p.add();
	cout << endl;
	p.subs();
	cout << endl;
	p.prod();

	return 0;
}
