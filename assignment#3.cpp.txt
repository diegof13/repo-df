#include<iostream>

using namespace std;

class Polynomial {
	int* Poly1, * Poly2, * Poly3, n;
public:
	Polynomial() {//Default
		Poly1 = new int[n];
		Poly2 = new int[n];
		Poly3 = new int[n];
	};
	//Polynomial(int arr1, int arr2, int arr3) {//Parametrized
		
	//};
	//Polynomial() {//Copy

	//};

	void ask_data() {//Asking data
		cout << "Enter size of polynomials: "; cin >> n;
		cout << "  (1ST POLYNOMIAL)";
		for (int i = 0;i < n;i++) {
			cout << "  \nEnter coefficient [" << i << "]: "; cin >> Poly1[i];
		}
		cout << "  (2ND POLYNOMIAL)";
		for (int i = 0;i < n;i++) {
			cout << "  \nEnter coefficient [" << i << "]: "; cin >> Poly2[i];
		}
	}
	void show_polynomials() {//showing polynomials
		cout << "P1 = ";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly1[i] << " ";
		}cout << endl;
		cout << "P2 = ";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly2[i] << " ";
		}
	}
	void addition() {
		for (int i = 0;i < n;i++) {
			Poly3[i] = Poly1[i] + Poly2[i];
		}cout << endl;
		cout << "p1+p2=";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly3[i] << " ";
		}
	}
	void subtraction() {
		for (int i = 0;i < n;i++) {
			Poly3[i] = Poly1[i] - Poly2[i];
		}cout << endl;
		cout << "p1-p2=";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly3[i] << " ";
		}
	}
	void product() {
		for (int i = 0;i < n;i++) {
			for (int j = 0;j < n;j++) {
				Poly3[i + j] = Poly1[i] * Poly2[j];
			}
		}cout << "\nProduct= ";
		for (int i = 0;i < n;i++) {
			for (int j = 0;j < n;j++) {
				cout << "  " << Poly3[i]<<" ";
			}
		}
	}
	~Polynomial() {
		delete[] Poly1;
		Poly1 = NULL;
		delete[] Poly2;
		Poly2 = NULL;
		delete[] Poly3;
		Poly3 = NULL;
	}
};

int main() {
	Polynomial p;
	p.ask_data();
	p.show_polynomials();
	p.addition();
	p.subtraction();
	p.product();

	return 0;
}
