	void add() {
		for ( i = 0;i < n;i++) {
			arr3[i] = arr1[i] + arr2[i];
		}cout << "SUM= ";
		x = i - 1;
		for (j = x;j >= 0;j--) {
			if (j == 0) {
				cout << arr3[j];
				break;
			}
			cout << arr3[j] << "(x^" << x << ")" << "+";
			x--;
		}
	}
	void subs() {
		for ( i = 0;i < n;i++) {
			arr3[i] = arr1[i] - arr2[i];
		}cout << "SUBS= ";
		x = i - 1;
		for (j = x;j >= 0;j--) {
			if (j == 0) {
				cout << arr3[j];
				break;
			}
			cout << arr3[j] << "(x^" << x << ")" << "+";
			x--;
		}
	}
	void prod() {
		cout << "PROD= ";
		for ( i = 0;i < n;i++) {
			for ( j = 0;j < m;j++) {
				arr3[i + j] = arr1[i] * arr2[j];
			}
		}
		for (i = 0;i < n;i++) {
			for (j = 0;j < m;j++) {
				cout << arr3[i + j] << " ";
			}
		}
	}
	Poly(Poly& f) {
