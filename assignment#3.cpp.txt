#include<iostream>

using namespace std;

class Polynomial {
	int n, Poly1[11], Poly2[11], Poly3[11];
public:
	void ask_data() {//Asking data
		cout << "Enter size of polynomials: "; cin >> n;
		cout << "  (1ST POLYNOMIAL)";
		for (int i = 0;i < n;i++) {
			cout << "  \nEnter coefficient [" << i << "]: "; cin >> Poly1[i];
		}
		cout << "  (2ND POLYNOMIAL)";
		for (int i = 0;i < n;i++) {
			cout << "  \nEnter coefficient [" << i << "]: "; cin >> Poly2[i];
		}
	}
	void show_polynomials() {//showing polynomials
		cout << "P1 = ";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly1[i] << " ";
		}cout << endl;
		cout << "P2 = ";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly2[i] << " ";
		}
	}
	void addition() {
		for (int i = 0;i < n;i++) {
			Poly3[i] = Poly1[i] + Poly2[i];
		}cout << endl;
		cout << "p1+p2=";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly3[i] << " ";
		}
	}
	void substraction() {
		for (int i = 0;i < n;i++) {
			Poly3[i] = Poly1[i] - Poly2[i];
		}cout << endl;
		cout << "p1-p2=";
		for (int i = 0;i < n;i++) {
			cout << "  " << Poly3[i] << " ";
		}
	}
};

int main() {
	Polynomial p;
	p.ask_data();
	p.show_polynomials();
	p.addition();
	p.substraction();
	//p.show_data();

	return 0;
}
