#include<iostream>

using namespace std;
    int is_year(int year) {
		if (year % 400 == 0) {

			cout << "This is a leap year.";
		}
		else {
			if (year % 4 == 0 and year %100 !=0) {
				cout << "This is a leap year.";
			}
			else {
				cout << "This is not a leap year.";
			}
		}return year;
}

int main() {
	int y;
	cout << "Enter the year: "; cin >> y;
	is_year(y);

}
